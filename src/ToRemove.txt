import java.util.ArrayList;
import java.util.List;

class Transaction {
    String description;
    double amount;

    public Transaction(String description, double amount) {
        this.description = description;
        this.amount = amount;
    }
}

class Category extends Transaction {
    double categoryTotal;

    public Category(String description) {
        super(description, 0);
        this.categoryTotal = 0;
    }

    public void addTransaction(double amount) {
        this.categoryTotal += amount;
    }
}

class Expense extends Transaction {
    Category associatedCategory;

    public Expense(String description, double amount, Category associatedCategory) {
        super(description, amount);
        this.associatedCategory = associatedCategory;
    }
}

class Summary {
    double totalAmountSpent;

    public Summary() {
        this.totalAmountSpent = 0;
    }

    public void updateTotalAmount(double amount) {
        this.totalAmountSpent += amount;
    }
}

class ExpenseTracker {
    List<Category> categoryList;
    List<Expense> expenseList;
    Summary summary;

    public ExpenseTracker() {
        this.categoryList = new ArrayList<>();
        this.expenseList = new ArrayList<>();
        this.summary = new Summary();
    }

    public void addCategory(Category category) {
        categoryList.add(category);
    }

    public void addExpense(Expense expense) {
        expenseList.add(expense);
        summary.updateTotalAmount(expense.amount);

        Category associatedCategory = expense.associatedCategory;
        if (associatedCategory != null) {
            associatedCategory.addTransaction(expense.amount);
        }
    }

    public void displayCategories() {
        for (Category category : categoryList) {
            System.out.println("Category: " + category.description + ", Total: " + category.categoryTotal);
        }
    }

    public void displayOverallSummary() {
        System.out.println("Overall Summary - Total Amount Spent: " + summary.totalAmountSpent);
    }
}

public class Main {
    public static void main(String[] args) {
        ExpenseTracker expenseTracker = new ExpenseTracker();

        // Adding categories
        Category category1 = new Category("Education");
        Category category2 = new Category("Entertainment");

        expenseTracker.addCategory(category1);
        expenseTracker.addCategory(category2);

        // Adding expenses with associated categories
        Expense expense1 = new Expense("Bought Books", 50.0, category1);
        Expense expense2 = new Expense("Movie Night", 30.0, category2);

        expenseTracker.addExpense(expense1);
        expenseTracker.addExpense(expense2);

        // Displaying categories and overall summary
        expenseTracker.displayCategories();
        expenseTracker.displayOverallSummary();
    }
}
